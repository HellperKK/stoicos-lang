(import :Loop)

; Tells if x is a divisor of y
(deffun :is_divisor [:x :y] {
  (== 0 (% y x))
})


; Tells if x is prime
(deffun :is_prime [:x] {
  (deffun :aux [:i] {
    (cond [
      [{(> (* i i) x)} true]
      [{(is_divisor i x)} false]
      [true {(aux (+ i 1))}]
    ])
  })
  (aux 2)
})


(Loop.upto_fun 2 100 (fun [:x] :(if (is_prime x) :(println x) unit)))
