(import :Game)
(import :Struct)

(def :settings (Struct.make [
  [:title "My first text game"]
  [:width 640]
  [:height 480]
  [:initial_state 0]
  [:update (fun [:state :_inputs] {(% (+ state 1) 640)})]
  [:draw (fun [:state] {
    [
      (Game.text 100 100 "Hello world")
    ]
  })]
]))

(Game.start settings)
